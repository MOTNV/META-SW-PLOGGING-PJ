export const rewardWeights = {
  collection: 4,
  coverage: 2,
  revisitPenalty: -1.5,
  overflowPenalty: -3,
};

export function calculateReward(metrics) {
  return (
    metrics.collected * rewardWeights.collection +
    metrics.coveredCells * rewardWeights.coverage +
    metrics.revisitCount * rewardWeights.revisitPenalty +
    metrics.overflowCount * rewardWeights.overflowPenalty
  );
}

export function summarizeReward(metrics) {
  return {
    reward: calculateReward(metrics),
    collected: metrics.collected,
    coveredCells: metrics.coveredCells,
    revisitCount: metrics.revisitCount,
    overflowCount: metrics.overflowCount,
  };
}
